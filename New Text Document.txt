Import-Csv "C:\path\to\your\file.csv" | ForEach-Object {
    $groupName = $_."Group Name"
    $result = if (Get-ADGroup -Filter { Name -eq $groupName }) {
        $members = Get-ADGroupMember -Identity $groupName | Select-Object -ExpandProperty SamAccountName -First 5
        $members -join ', ' -or 'No member found'
    } else {
        'No group found'
    }

    # Add the "Result" property dynamically
    $_ | Add-Member -MemberType NoteProperty -Name "Result" -Value $result -Force
    $_
} | Export-Csv "C:\path\to\your\file.csv" -NoTypeInformation
